def create
  @admin = Admin.new(admin_params)

  if @admin.save
    render json: @admin, status: :created
  else
    render json: { errors: @admin.errors }, status: :unprocessable_entity
  end
end

    Cria um novo administrador no sistema

    Argumentos¶ ↑

    Retorno¶ ↑
